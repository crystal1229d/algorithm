```js
/**
 * @param {number[]} arr
 * @param {Function} fn
 * @return {number[]}
 */

const filter = (arr, fn) => {
    let filteredArr = [];

    for(let i=0; i<arr.length; i++) {
        const result = fn(arr[i], i);
        if (result) filteredArr.push(arr[i]);
    }

    return filteredArr;
}
```
